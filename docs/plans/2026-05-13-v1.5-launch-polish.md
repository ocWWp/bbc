# BBC v1.5 Launch Polish Implementation Plan

**Goal:** Ship BBC v1.5 with: pre-launch security baseline + landing rewrite that passes the IQ-80 founder test + role-aware navigation that lets invited teammates do their job in <30 seconds + role-shaped Studios + admin Home dashboard + Inbox + clickable citation chips + Loop-3 visibility flag.

**Architecture:** Persona-aware UI on top of existing Next.js 16 / Supabase / Cloudflare stack. Teammate sees 3 routes (`/studio/<role>`, `/brain` read-only, `/inbox`). Admin sees full nav (adds `/home`, `/memory` editable, `/queue`, `/library`, `/settings`). Shared Studio skeleton + per-role chrome. Citation chips link to read-only `/brain/<id>`. Inbox has two channels (mentions / from-BBC). Security baseline is config + CI workflows + one threat-model doc — no new code to maintain. See `docs/plans/2026-05-13-bbc-persona-nav-design.md` for the full design.

**Tech Stack:** Next.js 16, React 19, Supabase (Postgres + RLS + Auth), Cloudflare Workers (via OpenNext), Anthropic Claude SDK, BlockNote editor, Tailwind. Security tools (off-the-shelf): Cloudflare WAF, Semgrep CI, Socket, Dependabot, OpenSSF Scorecard.

### Task 0a: RBAC model + ADR-0010 + page authorization

**Context:** `tenant_members.role` is the Postgres ENUM `public.tenant_role` with values `('admin', 'member', 'viewer')` (per `0003_tenant_model.sql`). Existing RLS (`0026_memory_files_write_policies.sql`) and `accept_proposal()` (`0008_write_path_functions.sql`) treat `member` as full editor. Hiding `/memory` and `/queue` from the nav does not stop a teammate from POSTing to `/memory/<id>` server actions or hitting RPCs directly. Currently `requireRole(a.actor, "member")` in `settings/keys/actions.ts` (and others) gates "any authenticated member of the tenant", which a new read-only member would pass.

**Decision (after ADR-0010):** add a new enum value `operator` between `admin` and `member`. Map:

- `admin` — full nav, manages teammates + tenant settings, all writes.
- `operator` — what `member` is today. Edits memory, accept/reject queue, sees all routes. Power users next to the admin. **Existing `member` rows migrate to `operator` so live tenants don't lose write power.**
- `member` (new semantics) — read-only on memory (`/brain` only, no `/memory` page access), can file `propose_change` proposals and run their Studio. Cannot accept/reject queue. Cannot manage settings.
- `viewer` — pre-existing role; treat as identical to new `member` for now (no UI flow currently creates viewers). Document for future cleanup.

Alternatives considered: separate `tenant_member_permissions` table (rejected — too much surface for a launch). Capability-per-feature flags (rejected — adds permission sprawl).

#### Step 2: Migration — ENUM ADD VALUE (not check constraint)

`apps/dashboard/supabase/migrations/0037_rbac_operator_role.sql`:

```sql
-- Add 'operator' to the tenant_role enum.
-- Existing 'member' rows migrate to 'operator' so live tenants don't lose write power.
-- New 'member' semantics: read-only on memory, can file proposals.

-- Postgres enum modification must be in its own transaction-level statement.
alter type public.tenant_role add value if not exists 'operator';

-- After the enum is committed, migrate existing data. Run this in a SEPARATE migration
-- file if the executor complains about "unsafe use of new value of enum type" in the
-- same transaction. If so, name this 0037b_rbac_member_migrate.sql.
update public.tenant_members set role = 'operator'::public.tenant_role
  where role = 'member'::public.tenant_role;

-- Helper: is_operator_of(tenant) — admin OR operator.
-- Mirrors the hardening pattern used by is_member_of in 0003_tenant_model.sql:
-- security definer, fixed search_path, execute revoked from public/anon/authenticated
-- so the function can only be called from RLS policies and server-role contexts.
create or replace function public.is_operator_of(p_tenant_id uuid)
returns boolean
language sql
stable
security definer
set search_path = public, auth
as $$
  select exists (
    select 1 from public.tenant_members
    where tenant_id = p_tenant_id
      and user_id = auth.uid()
      and role in ('admin'::public.tenant_role, 'operator'::public.tenant_role)
  );
$$;
revoke execute on function public.is_operator_of(uuid) from public, anon, authenticated;

-- Update memory_files RLS: members SELECT-only; only operators+admins write.
drop policy if exists memory_files_member_insert on public.memory_files;
drop policy if exists memory_files_member_update on public.memory_files;
drop policy if exists memory_files_member_delete on public.memory_files;

create policy memory_files_operator_insert on public.memory_files
  for insert with check (public.is_operator_of(tenant_id));
create policy memory_files_operator_update on public.memory_files
  for update using (public.is_operator_of(tenant_id));
create policy memory_files_operator_delete on public.memory_files
  for delete using (public.is_operator_of(tenant_id));
-- Members keep SELECT via existing memory_files_member_select policy (unchanged).
```

> **Enum ADD VALUE caveat:** Postgres requires the enum modification to commit before new values can be used in DML. If the migration runner runs all of Step 2 as one transaction and errors with "unsafe use of new value of enum type", split into `0037a_rbac_operator_enum.sql` (just the `ALTER TYPE`) and `0037b_rbac_operator_migrate.sql` (the `UPDATE` + policies). Verify against the Supabase migration runner before running.

#### Step 4: Update `Role` type + `requireRole()`

In `apps/dashboard/src/lib/auth/require-user.ts`:

```ts
export type Role = "admin" | "operator" | "member" | "viewer";

// Rank: admin > operator > member > viewer.
const RANK: Record<Role, number> = { admin: 3, operator: 2, member: 1, viewer: 0 };

export function requireRole(actor: Actor, atLeast: Role): { ok: true } | { ok: false; error: string } {
  if (RANK[actor.role] >= RANK[atLeast]) return { ok: true };
  return { ok: false, error: `requires ${atLeast}` };
}
```

#### Step 1: SQL function (writes to `body` + `frontmatter` jsonb)

```sql
create or replace function public.propose_change(
  p_tenant_id        uuid,
  p_target_file      text,
  p_change_kind      text,
  p_summary          text,
  p_body             text,
  p_source_memory_id uuid default null
) returns text  -- returns the text proposal_id (slug), same shape as scripts/propose.sh
language plpgsql
security definer
set search_path = public
as $$
declare
  v_user_id     uuid := auth.uid();
  v_slug        text;
  v_proposal_id text;
  v_frontmatter jsonb;
begin
  if v_user_id is null then
    raise exception 'not authenticated';
  end if;
  if not public.is_member_of(p_tenant_id) then
    raise exception 'not a member of tenant';
  end if;
  if p_change_kind not in ('edit', 'add', 'supersede', 'archive', 'flag') then
    raise exception 'invalid change_kind: %', p_change_kind;
  end if;

  -- Generate a stable slug-based proposal_id, same format scripts/propose.sh uses.
  v_slug := substring(regexp_replace(lower(p_summary), '[^a-z0-9]+', '-', 'g'), 1, 40);
  v_slug := trim(both '-' from v_slug);
  v_proposal_id := format(
    'prop_%s_%s_%s',
    to_char(now() at time zone 'utc', 'YYYYMMDDHH24MISS'),
    v_slug,
    substring(gen_random_uuid()::text, 1, 6)
  );

  -- Frontmatter mirrors scripts/propose.sh YAML frontmatter shape.
  v_frontmatter := jsonb_build_object(
    'proposal_id',       v_proposal_id,
    'proposed_by',       v_user_id::text,
    'proposed_at',       now() at time zone 'utc',
    'target_layer',      'main',  -- or 'manager'; pass as param if needed
    'target_file',       p_target_file,
    'change_kind',       p_change_kind,
    'diff_summary',      p_summary,
    'source',            'dashboard',
    'source_memory_id',  p_source_memory_id
  );

  insert into public.queue_items (tenant_id, proposal_id, status, body, frontmatter)
  values (p_tenant_id, v_proposal_id, 'pending'::public.queue_status, p_body, v_frontmatter);

  return v_proposal_id;
end;
$$;

grant execute on function public.propose_change(uuid, text, text, text, text, uuid) to authenticated;
```

```ts
import { execFile } from "node:child_process";
import { promisify } from "node:util";
import { writeFile, mkdtemp } from "node:fs/promises";
import { join } from "node:path";
import { tmpdir } from "node:os";
const execFileAsync = promisify(execFile);

async fileProposal(input) {
  // Body to a temp file (propose.sh expects --body-file for non-trivial bodies).
  const tmp = await mkdtemp(join(tmpdir(), "bbc-prop-"));
  const bodyPath = join(tmp, "body.md");
  await writeFile(bodyPath, input.body, "utf8");

  const args = [
    "scripts/propose.sh",
    "--target", "main",
    // propose.sh errors if --originator is missing AND PWD is repo root (its inference
    // expects PWD inside distribution/<leaf>/ or manager/). LocalStore runs server-side
    // and writes from repo root, so pass the originator explicitly.
    "--originator", "leaf-dashboard",
    "--file", input.target_file,
    "--kind", input.change_kind,
    "--summary", input.summary,  // execFile passes arg as a single argv slot — no quoting needed.
    "--body-file", bodyPath,
  ];
  if (input.source_memory_id) args.push("--source-memory-id", input.source_memory_id);

  const { stdout } = await execFileAsync("bash", args, { cwd: this.bbcRoot });
  const match = stdout.match(/proposal_id:\s*(\S+)/);
  if (!match) throw new Error("could not parse proposal_id from propose.sh output");
  return { id: match[1] };
}
```

```ts
// apps/dashboard/src/lib/studio/template-id.ts
export const ROLE_PREFIXES = {
  marketing: "marketing:",
  engineering: "eng:",
  founder: "founder:",
  designer: "design:",
  support: "support:",
} as const;

export type Role = keyof typeof ROLE_PREFIXES;

export function roleForTemplateId(templateId: string): Role | null {
  for (const [role, prefix] of Object.entries(ROLE_PREFIXES) as Array<[Role, string]>) {
    if (templateId.startsWith(prefix)) return role;
  }
  return null;
}

export function templateIdsForRole(role: Role): string {
  // Postgres LIKE pattern: e.g. 'marketing:%' for the role's runs.
  return `${ROLE_PREFIXES[role]}%`;
}
```

```tsx
import { redirect } from "next/navigation";
import { requireActor } from "@/lib/auth/require-user";
import { getSupabaseServerClient } from "@/lib/supabase/server";

export const dynamic = "force-dynamic";

export default async function Root() {
  const a = await requireActor();
  if (!a.ok) redirect("/queue");

  // Preserve the empty-brain → /welcome gate. Do not skip this — fresh tenants
  // routing straight to /home would land on a dashboard with nothing in it.
  const supabase = await getSupabaseServerClient();
  const { count } = await supabase
    .from("memory_files")
    .select("id", { count: "exact", head: true })
    .eq("tenant_id", a.actor.tenant_id);
  if ((count ?? 0) === 0) redirect("/welcome");

  if (a.actor.role === "admin") redirect("/home");
  const roleSlug = (a.actor.templateSlug ?? "marketing").toLowerCase();
  redirect(`/studio/${roleSlug}`);
}
```

```ts
"use server";

import { requireActor } from "@/lib/auth/require-user";
import { getStore } from "@/lib/store";
import { getMemoryItem } from "@/app/memory/queries";

export async function flagMemory(formData: FormData): Promise<{ ok: boolean; id?: string; code?: string }> {
  const a = await requireActor();
  if (!a.ok) return { ok: false, code: "unauthorized" };

  const memoryId = String(formData.get("memory_id") ?? "");
  const reason = String(formData.get("reason") ?? "");
  if (!memoryId || !reason) return { ok: false, code: "invalid_input" };

  // Look up the memory so we can target the right file path.
  const memory = await getMemoryItem(memoryId);
  if (!memory || memory.tenant_id !== a.actor.tenant_id) {
    return { ok: false, code: "not_found" };
  }

  // memory_files.path is already the relative path inside memory/, e.g. 'memory/design/voice-tone.md'.
  // If the path is missing for any reason (legacy rows), fall back to an id-keyed path.
  const targetFile = memory.path ?? `memory/files/${memoryId}.md`;

  const store = await getStore();
  const result = await store.queue.fileProposal({
    tenant_id: a.actor.tenant_id,
    target_file: targetFile,
    change_kind: "flag",
    summary: `Flag: ${reason.slice(0, 80)}`,
    body: `Flagged by ${a.actor.user_id} (member):\n\n${reason}\n\n— filed from /brain/${memoryId}`,
    source_memory_id: memoryId,
  });
  return { ok: true, id: result.id };
}
```

```sql
create table public.inbox_items (
  id uuid primary key default gen_random_uuid(),
  tenant_id uuid not null references public.tenants(id) on delete cascade,
  user_id uuid not null references auth.users(id) on delete cascade,
  channel text not null check (channel in ('mentions', 'from_bbc')),
  kind text not null,  -- 'flag_resolved' | 'loop3_suggestion' | 'mention' (future)
  title text not null,
  body text,
  source_kind text,    -- 'queue_item' | 'recommendation' | 'memory_file' | null
  -- queue_items.id is uuid; queue_items.proposal_id is text (slug). Link by uuid id.
  source_queue_item_id uuid references public.queue_items(id) on delete set null,
  source_recommendation_id uuid references public.recommendations(id) on delete set null,
  source_memory_id uuid references public.memory_files(id) on delete set null,
  flagger_user_id uuid references auth.users(id) on delete set null,
  read_at timestamptz,
  created_at timestamptz not null default now()
);

create index inbox_items_user_channel_unread_idx
  on public.inbox_items (user_id, channel, read_at, created_at desc);

alter table public.inbox_items enable row level security;

-- SELECT: the owner of the row
create policy inbox_items_owner_select on public.inbox_items
  for select using (user_id = auth.uid() and public.is_member_of(tenant_id));

-- UPDATE: owner can mark their own row read. The trigger below enforces that ONLY
-- read_at may change — not body/title/source links — so the client SDK cannot
-- mutate notification content.
create policy inbox_items_owner_update on public.inbox_items
  for update using (user_id = auth.uid() and public.is_member_of(tenant_id))
  with check (user_id = auth.uid() and public.is_member_of(tenant_id));

create or replace function public.inbox_items_only_read_at_changes()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
begin
  if  (old.tenant_id is distinct from new.tenant_id)
   or (old.user_id   is distinct from new.user_id)
   or (old.channel   is distinct from new.channel)
   or (old.kind      is distinct from new.kind)
   or (old.title     is distinct from new.title)
   or (old.body      is distinct from new.body)
   or (old.source_kind is distinct from new.source_kind)
   or (old.source_queue_item_id     is distinct from new.source_queue_item_id)
   or (old.source_recommendation_id is distinct from new.source_recommendation_id)
   or (old.source_memory_id is distinct from new.source_memory_id)
   or (old.flagger_user_id  is distinct from new.flagger_user_id)
   or (old.created_at is distinct from new.created_at)
  then
    raise exception 'inbox_items: only read_at may be updated by the owner';
  end if;
  return new;
end
$$;

create trigger inbox_items_lock_fields
  before update on public.inbox_items
  for each row execute function public.inbox_items_only_read_at_changes();

-- INSERT: only via service-role context (server actions / RPCs).
-- No member-level insert policy on purpose. Inserts come from:
--   1) queue resolution hook (Task 32) writing as the service role
--   2) Loop-3 fan-out cron / job (post-launch) writing as the service role
-- If we need an end-user-driven insert path later (e.g. mentions), add a separate policy.

-- DELETE: not allowed at RLS layer in v1.5.
```

```ts
import { getSupabaseServiceClient } from "@/lib/supabase/server";

export type InboxInsert = {
  tenant_id: string;
  user_id: string;
  channel: "mentions" | "from_bbc";
  kind: "flag_resolved" | "loop3_suggestion" | "mention";
  title: string;
  body?: string;
  source_kind?: "queue_item" | "recommendation" | "memory_file";
  source_queue_item_id?: string;     // uuid (queue_items.id, not proposal_id text slug)
  source_recommendation_id?: string; // uuid
  source_memory_id?: string;         // uuid
  flagger_user_id?: string;
};

export async function insertInboxItem(item: InboxInsert) {
  const supabase = getSupabaseServiceClient();  // service role bypasses RLS — only callable from server actions
  const { data, error } = await supabase.from("inbox_items").insert(item).select("id").single();
  if (error) throw error;
  return data.id;
}
```
